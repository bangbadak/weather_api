49   def _match_vcs_scheme(url):
50       # type: (str) -> Optional[str]
51       """Look for VCS schemes in the URL. 
52    
53       Returns the matched VCS scheme, or None if there's no match. 
54       """
55       for scheme in vcs.schemes:
56           if url.lower().startswith(scheme) and url[len(scheme)] in '+:':
57               return scheme
58       return None

69   def _ensure_html_header(response):
70       # type: (Response) -> None
71       """Check the Content-Type header to ensure the response contains HTML. 
72    
73       Raises `_NotHTML` if the content type is not text/html. 
74       """
75       content_type = response.headers.get("Content-Type", "")
76       if not content_type.lower().startswith("text/html"):
77           raise _NotHTML(content_type, response.request.method)

84   def _ensure_html_response(url, session):
85       # type: (str, PipSession) -> None
86       """Send a HEAD request to the URL, and ensure the response contains HTML. 
87    
88       Raises `_NotHTTP` if the URL is not available for a HEAD request, or 
89       `_NotHTML` if the content type is not text/html. 
90       """
91       scheme, netloc, path, query, fragment = urllib.parse.urlsplit(url)
92       if scheme not in {'http', 'https'}:
93           raise _NotHTTP()
94   
95       resp = session.head(url, allow_redirects=True)
96       raise_for_status(resp)
97   
98       _ensure_html_header(resp)

101  def _get_html_response(url, session):
102      # type: (str, PipSession) -> Response
103      """Access an HTML page with GET, and return the response. 
104   
105      This consists of three parts: 
106   
107      1. If the URL looks suspiciously like an archive, send a HEAD first to 
108         check the Content-Type is HTML, to avoid downloading a large file. 
109         Raise `_NotHTTP` if the content type cannot be determined, or 
110         `_NotHTML` if it is not HTML. 
111      2. Actually perform the request. Raise HTTP exceptions on network failures. 
112      3. Check the Content-Type header to make sure we got HTML, and raise 
113         `_NotHTML` otherwise. 
114      """
115      if is_archive_file(Link(url).filename):
116          _ensure_html_response(url, session=session)
117  
118      logger.debug('Getting page %s', redact_auth_from_url(url))
119  
120      resp = session.get(
121          url,
122          headers={
123              "Accept": "text/html",
124              # We don't want to blindly returned cached data for
125              # /simple/, because authors generally expecting that
126              # twine upload && pip install will function, but if
127              # they've done a pip install in the last ~10 minutes
128              # it won't. Thus by setting this to zero we will not
129              # blindly use any cached data, however the benefit of
130              # using max-age=0 instead of no-cache, is that we will
131              # still support conditional requests, so we will still
132              # minimize traffic sent in cases where the page hasn't
133              # changed at all, we will just always incur the round
134              # trip for the conditional GET now instead of only
135              # once per 10 minutes.
136              # For more information, please see pypa/pip#5670.
137              "Cache-Control": "max-age=0",
138          },
139      )
140      raise_for_status(resp)
141  
142      # The check for archives above only works if the url ends with
143      # something that looks like an archive. However that is not a
144      # requirement of an url. Unless we issue a HEAD request on every
145      # url we cannot know ahead of time for sure if something is HTML
146      # or not. However we can check after we've downloaded it.
147      _ensure_html_header(resp)
148  
149      return resp

152  def _get_encoding_from_headers(headers):
153      # type: (ResponseHeaders) -> Optional[str]
154      """Determine if we have any encoding information in our headers. 
155      """
156      if headers and "Content-Type" in headers:
157          content_type, params = cgi.parse_header(headers["Content-Type"])
158          if "charset" in params:
159              return params['charset']
160      return None

163  def _determine_base_url(document, page_url):
164      # type: (HTMLElement, str) -> str
165      """Determine the HTML document's base URL. 
166   
167      This looks for a ``<base>`` tag in the HTML document. If present, its href 
168      attribute denotes the base URL of anchor tags in the document. If there is 
169      no such tag (or if it does not have a valid href attribute), the HTML 
170      file's URL is used as the base URL. 
171   
172      :param document: An HTML document representation. The current 
173          implementation expects the result of ``html5lib.parse()``. 
174      :param page_url: The URL of the HTML document. 
175      """
176      for base in document.findall(".//base"):
177          href = base.get("href")
178          if href is not None:
179              return href
180      return page_url

183  def _clean_url_path_part(part):
184      # type: (str) -> str
185      """ 
186      Clean a "part" of a URL path (i.e. after splitting on "@" characters). 
187      """
188      # We unquote prior to quoting to make sure nothing is double quoted.
189      return urllib.parse.quote(urllib.parse.unquote(part))

192  def _clean_file_url_path(part):
193      # type: (str) -> str
194      """ 
195      Clean the first part of a URL path that corresponds to a local 
196      filesystem path (i.e. the first part after splitting on "@" characters). 
197      """
198      # We unquote prior to quoting to make sure nothing is double quoted.
199      # Also, on Windows the path part might contain a drive letter which
200      # should not be quoted. On Linux where drive letters do not
201      # exist, the colon should be quoted. We rely on urllib.request
202      # to do the right thing here.
203      return urllib.request.pathname2url(urllib.request.url2pathname(part))

206  # percent-encoded:                   /
207  _reserved_chars_re = re.compile('(@|%2F)', re.IGNORECASE)
208  
209  
210  def _clean_url_path(path, is_local_path):
211      # type: (str, bool) -> str
212      """ 
213      Clean the path portion of a URL. 
214      """
215      if is_local_path:
216          clean_func = _clean_file_url_path
217      else:
218          clean_func = _clean_url_path_part
219  
220      # Split on the reserved characters prior to cleaning so that
221      # revision strings in VCS URLs are properly preserved.
222      parts = _reserved_chars_re.split(path)
223  
224      cleaned_parts = []
225      for to_clean, reserved in pairwise(itertools.chain(parts, [''])):
226          cleaned_parts.append(clean_func(to_clean))
227          # Normalize %xx escapes (e.g. %2f -> %2F)
228          cleaned_parts.append(reserved.upper())
229  
230      return ''.join(cleaned_parts)

233  def _clean_link(url):
234      # type: (str) -> str
235      """ 
236      Make sure a link is fully quoted. 
237      For example, if ' ' occurs in the URL, it will be replaced with "%20", 
238      and without double-quoting other characters. 
239      """
240      # Split the URL into parts according to the general structure
241      # `scheme://netloc/path;parameters?query#fragment`.
242      result = urllib.parse.urlparse(url)
243      # If the netloc is empty, then the URL refers to a local filesystem path.
244      is_local_path = not result.netloc
245      path = _clean_url_path(result.path, is_local_path=is_local_path)
246      return urllib.parse.urlunparse(result._replace(path=path))

249  def _create_link_from_element(
250      anchor,    # type: HTMLElement
251      page_url,  # type: str
252      base_url,  # type: str
253  ):
254      # type: (...) -> Optional[Link]
255      """ 
256      Convert an anchor element in a simple repository page to a Link. 
257      """
258      href = anchor.get("href")
259      if not href:
260          return None
261  
262      url = _clean_link(urllib.parse.urljoin(base_url, href))
263      pyrequire = anchor.get('data-requires-python')
264      pyrequire = html.unescape(pyrequire) if pyrequire else None
265  
266      yanked_reason = anchor.get('data-yanked')
267      if yanked_reason:
268          yanked_reason = html.unescape(yanked_reason)
269  
270      link = Link(
271          url,
272          comes_from=page_url,
273          requires_python=pyrequire,
274          yanked_reason=yanked_reason,
275      )
276  
277      return link

296  def with_cached_html_pages(
297      fn,    # type: Callable[[HTMLPage], Iterable[Link]]
298  ):
299      # type: (...) -> Callable[[HTMLPage], List[Link]]
300      """ 
301      Given a function that parses an Iterable[Link] from an HTMLPage, cache the 
302      function's result (keyed by CacheablePageContent), unless the HTMLPage 
303      `page` has `page.cache_link_parsing == False`. 
304      """
305  
306      @functools.lru_cache(maxsize=None)
307      def wrapper(cacheable_page):
308          # type: (CacheablePageContent) -> List[Link]
309          return list(fn(cacheable_page.page))
310  
311      @functools.wraps(fn)
312      def wrapper_wrapper(page):
313          # type: (HTMLPage) -> List[Link]
314          if page.cache_link_parsing:
315              return wrapper(CacheablePageContent(page))
316          return list(fn(page))
317  
318      return wrapper_wrapper

321  @with_cached_html_pages
322  def parse_links(page):
323      # type: (HTMLPage) -> Iterable[Link]
324      """ 
325      Parse an HTML document, and yield its anchor elements as Link objects. 
326      """
327      document = html5lib.parse(
328          page.content,
329          transport_encoding=page.encoding,
330          namespaceHTMLElements=False,
331      )
332  
333      url = page.url
334      base_url = _determine_base_url(document, url)
335      for anchor in document.findall(".//a"):
336          link = _create_link_from_element(
337              anchor,
338              page_url=url,
339              base_url=base_url,
340          )
341          if link is None:
342              continue
343          yield link

395  def _get_html_page(link, session=None):
396      # type: (Link, Optional[PipSession]) -> Optional[HTMLPage]
397      if session is None:
398          raise TypeError(
399              "_get_html_page() missing 1 required keyword argument: 'session'"
400          )
401  
402      url = link.url.split('#', 1)[0]
403  
404      # Check for VCS schemes that do not support lookup as web pages.
405      vcs_scheme = _match_vcs_scheme(url)
406      if vcs_scheme:
407          logger.warning('Cannot look at %s URL %s because it does not support '
408                         'lookup as web pages.', vcs_scheme, link)
409          return None
410  
411      # Tack index.html onto file:// URLs that point to directories
412      scheme, _, path, _, _, _ = urllib.parse.urlparse(url)
413      if (scheme == 'file' and os.path.isdir(urllib.request.url2pathname(path))):
414          # add trailing slash if not present so urljoin doesn't trim
415          # final segment
416          if not url.endswith('/'):
417              url += '/'
418          url = urllib.parse.urljoin(url, 'index.html')
419          logger.debug(' file: URL is directory, getting %s', url)
420  
421      try:
422          resp = _get_html_response(url, session=session)
423      except _NotHTTP:
424          logger.warning(
425              'Skipping page %s because it looks like an archive, and cannot '
426              'be checked by a HTTP HEAD request.', link,
427          )
428      except _NotHTML as exc:
429          logger.warning(
430              'Skipping page %s because the %s request got Content-Type: %s.'
431              'The only supported Content-Type is text/html',
432              link, exc.request_desc, exc.content_type,
433          )
434      except NetworkConnectionError as exc:
435          _handle_get_page_fail(link, exc)
436      except RetryError as exc:
437          _handle_get_page_fail(link, exc)
438      except SSLError as exc:
439          reason = "There was a problem confirming the ssl certificate: "
440          reason += str(exc)
441          _handle_get_page_fail(link, reason, meth=logger.info)
442      except requests.ConnectionError as exc:
443          _handle_get_page_fail(link, f"connection error: {exc}")
444      except requests.Timeout:
445          _handle_get_page_fail(link, "timed out")
446      else:
447          return _make_html_page(resp,
448                                 cache_link_parsing=link.cache_link_parsing)
449      return None

457  class LinkCollector:
458  
459      """ 
460      Responsible for collecting Link objects from all configured locations, 
461      making network requests as needed. 
462   
463      The class's main method is its collect_sources() method. 
464      """
465  
466      def __init__(
467          self,
468          session,       # type: PipSession
469          search_scope,  # type: SearchScope
470      ):
471          # type: (...) -> None
472          self.search_scope = search_scope
473          self.session = session
474  
475      @classmethod
476      def create(cls, session, options, suppress_no_index=False):
477          # type: (PipSession, Values, bool) -> LinkCollector
478          """ 
479          :param session: The Session to use to make requests. 
480          :param suppress_no_index: Whether to ignore the --no-index option 
481              when constructing the SearchScope object. 
482          """
483          index_urls = [options.index_url] + options.extra_index_urls
484          if options.no_index and not suppress_no_index:
485              logger.debug(
486                  'Ignoring indexes: %s',
487                  ','.join(redact_auth_from_url(url) for url in index_urls),
488              )
489              index_urls = []
490  
491          # Make sure find_links is a list before passing to create().
492          find_links = options.find_links or []
493  
494          search_scope = SearchScope.create(
495              find_links=find_links, index_urls=index_urls,
496          )
497          link_collector = LinkCollector(
498              session=session, search_scope=search_scope,
499          )
500          return link_collector
501  
502      @property
503      def find_links(self):
504          # type: () -> List[str]
505          return self.search_scope.find_links
506  
507      def fetch_page(self, location):
508          # type: (Link) -> Optional[HTMLPage]
509          """ 
510          Fetch an HTML page containing package links. 
511          """
512          return _get_html_page(location, session=self.session)
513  
514      def collect_sources(
515          self,
516          project_name: str,
517          candidates_from_page: CandidatesFromPage,
518      ) -> CollectedSources:
519          # The OrderedDict calls deduplicate sources by URL.
520          index_url_sources = collections.OrderedDict(
521              build_source(
522                  loc,
523                  candidates_from_page=candidates_from_page,
524                  page_validator=self.session.is_secure_origin,
525                  expand_dir=False,
526                  cache_link_parsing=False,
527              )
528              for loc in self.search_scope.get_index_urls_locations(project_name)
529          ).values()
530          find_links_sources = collections.OrderedDict(
531              build_source(
532                  loc,
533                  candidates_from_page=candidates_from_page,
534                  page_validator=self.session.is_secure_origin,
535                  expand_dir=True,
536                  cache_link_parsing=True,
537              )
538              for loc in self.find_links
539          ).values()
540  
541          if logger.isEnabledFor(logging.DEBUG):
542              lines = [
543                  f"* {s.link}"
544                  for s in itertools.chain(find_links_sources, index_url_sources)
545                  if s is not None and s.link is not None
546              ]
547              lines = [
548                  f"{len(lines)} location(s) to search "
549                  f"for versions of {project_name}:"
550              ] + lines
551              logger.debug("\n".join(lines))
552  
553          return CollectedSources(
554              find_links=list(find_links_sources),
555              index_urls=list(index_url_sources),
556          )
