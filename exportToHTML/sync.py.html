21   def _restore_context(context):
22       # Check for changes in contextvars, and set them to the current
23       # context for downstream consumers
24       for cvar in context:
25           try:
26               if cvar.get() != context.get(cvar):
27                   cvar.set(context.get(cvar))
28           except LookupError:
29               cvar.set(context.get(cvar))

32   def _iscoroutinefunction_or_partial(func: Any) -> bool:
33       # Python < 3.8 does not correctly determine partially wrapped
34       # coroutine functions are coroutine functions, hence the need for
35       # this to exist. Code taken from CPython.
36       if sys.version_info >= (3, 8):
37           return asyncio.iscoroutinefunction(func)
38       else:
39           while inspect.ismethod(func):
40               func = func.__func__
41           while isinstance(func, functools.partial):
42               func = func.func
43   
44           return asyncio.iscoroutinefunction(func)

47   class ThreadSensitiveContext:
48       """Async context manager to manage context for thread sensitive mode 
49    
50       This context manager controls which thread pool executor is used when in 
51       thread sensitive mode. By default, a single thread pool executor is shared 
52       within a process. 
53    
54       In Python 3.7+, the ThreadSensitiveContext() context manager may be used to 
55       specify a thread pool per context. 
56    
57       In Python 3.6, usage of this context manager has no effect. 
58    
59       This context manager is re-entrant, so only the outer-most call to 
60       ThreadSensitiveContext will set the context. 
61    
62       Usage: 
63    
64       >>> import time 
65       >>> async with ThreadSensitiveContext(): 
66       ...     await sync_to_async(time.sleep, 1)() 
67       """
68   
69       def __init__(self):
70           self.token = None
71   
72       if contextvars:
73   
74           async def __aenter__(self):
75               try:
76                   SyncToAsync.thread_sensitive_context.get()
77               except LookupError:
78                   self.token = SyncToAsync.thread_sensitive_context.set(self)
79   
80               return self
81   
82           async def __aexit__(self, exc, value, tb):
83               if not self.token:
84                   return
85   
86               executor = SyncToAsync.context_to_thread_executor.pop(self, None)
87               if executor:
88                   executor.shutdown()
89               SyncToAsync.thread_sensitive_context.reset(self.token)
90   
91       else:
92   
93           async def __aenter__(self):
94               return self
95   
96           async def __aexit__(self, exc, value, tb):
97               pass

100  class AsyncToSync:
101      """ 
102      Utility class which turns an awaitable that only works on the thread with 
103      the event loop into a synchronous callable that works in a subthread. 
104   
105      If the call stack contains an async loop, the code runs there. 
106      Otherwise, the code runs in a new loop in a new thread. 
107   
108      Either way, this thread then pauses and waits to run any thread_sensitive 
109      code called from further down the call stack using SyncToAsync, before 
110      finally exiting once the async task returns. 
111      """
112  
113      # Maps launched Tasks to the threads that launched them (for locals impl)
114      launch_map: "Dict[asyncio.Task[object], threading.Thread]" = {}
115  
116      # Keeps track of which CurrentThreadExecutor to use. This uses an asgiref
117      # Local, not a threadlocal, so that tasks can work out what their parent used.
118      executors = Local()
119  
120      def __init__(self, awaitable, force_new_loop=False):
121          if not callable(awaitable) or not _iscoroutinefunction_or_partial(awaitable):
122              # Python does not have very reliable detection of async functions
123              # (lots of false negatives) so this is just a warning.
124              warnings.warn("async_to_sync was passed a non-async-marked callable")
125          self.awaitable = awaitable
126          try:
127              self.__self__ = self.awaitable.__self__
128          except AttributeError:
129              pass
130          if force_new_loop:
131              # They have asked that we always run in a new sub-loop.
132              self.main_event_loop = None
133          else:
134              try:
135                  self.main_event_loop = asyncio.get_event_loop()
136              except RuntimeError:
137                  # There's no event loop in this thread. Look for the threadlocal if
138                  # we're inside SyncToAsync
139                  main_event_loop_pid = getattr(
140                      SyncToAsync.threadlocal, "main_event_loop_pid", None
141                  )
142                  # We make sure the parent loop is from the same process - if
143                  # they've forked, this is not going to be valid any more (#194)
144                  if main_event_loop_pid and main_event_loop_pid == os.getpid():
145                      self.main_event_loop = getattr(
146                          SyncToAsync.threadlocal, "main_event_loop", None
147                      )
148                  else:
149                      self.main_event_loop = None
150  
151      def __call__(self, *args, **kwargs):
152          # You can't call AsyncToSync from a thread with a running event loop
153          try:
154              event_loop = asyncio.get_event_loop()
155          except RuntimeError:
156              pass
157          else:
158              if event_loop.is_running():
159                  raise RuntimeError(
160                      "You cannot use AsyncToSync in the same thread as an async event loop - "
161                      "just await the async function directly."
162                  )
163  
164          if contextvars is not None:
165              # Wrapping context in list so it can be reassigned from within
166              # `main_wrap`.
167              context = [contextvars.copy_context()]
168          else:
169              context = None
170  
171          # Make a future for the return information
172          call_result = Future()
173          # Get the source thread
174          source_thread = threading.current_thread()
175          # Make a CurrentThreadExecutor we'll use to idle in this thread - we
176          # need one for every sync frame, even if there's one above us in the
177          # same thread.
178          if hasattr(self.executors, "current"):
179              old_current_executor = self.executors.current
180          else:
181              old_current_executor = None
182          current_executor = CurrentThreadExecutor()
183          self.executors.current = current_executor
184          # Use call_soon_threadsafe to schedule a synchronous callback on the
185          # main event loop's thread if it's there, otherwise make a new loop
186          # in this thread.
187          try:
188              awaitable = self.main_wrap(
189                  args, kwargs, call_result, source_thread, sys.exc_info(), context
190              )
191  
192              if not (self.main_event_loop and self.main_event_loop.is_running()):
193                  # Make our own event loop - in a new thread - and run inside that.
194                  loop = asyncio.new_event_loop()
195                  loop_executor = ThreadPoolExecutor(max_workers=1)
196                  loop_future = loop_executor.submit(
197                      self._run_event_loop, loop, awaitable
198                  )
199                  if current_executor:
200                      # Run the CurrentThreadExecutor until the future is done
201                      current_executor.run_until_future(loop_future)
202                  # Wait for future and/or allow for exception propagation
203                  loop_future.result()
204              else:
205                  # Call it inside the existing loop
206                  self.main_event_loop.call_soon_threadsafe(
207                      self.main_event_loop.create_task, awaitable
208                  )
209                  if current_executor:
210                      # Run the CurrentThreadExecutor until the future is done
211                      current_executor.run_until_future(call_result)
212          finally:
213              # Clean up any executor we were running
214              if hasattr(self.executors, "current"):
215                  del self.executors.current
216              if old_current_executor:
217                  self.executors.current = old_current_executor
218              if contextvars is not None:
219                  _restore_context(context[0])
220  
221          # Wait for results from the future.
222          return call_result.result()
223  
224      def _run_event_loop(self, loop, coro):
225          """ 
226          Runs the given event loop (designed to be called in a thread). 
227          """
228          asyncio.set_event_loop(loop)
229          try:
230              loop.run_until_complete(coro)
231          finally:
232              try:
233                  # mimic asyncio.run() behavior
234                  # cancel unexhausted async generators
235                  if sys.version_info >= (3, 7, 0):
236                      tasks = asyncio.all_tasks(loop)
237                  else:
238                      tasks = asyncio.Task.all_tasks(loop)
239                  for task in tasks:
240                      task.cancel()
241                  loop.run_until_complete(asyncio.gather(*tasks, return_exceptions=True))
242                  for task in tasks:
243                      if task.cancelled():
244                          continue
245                      if task.exception() is not None:
246                          loop.call_exception_handler(
247                              {
248                                  "message": "unhandled exception during loop shutdown",
249                                  "exception": task.exception(),
250                                  "task": task,
251                              }
252                          )
253                  if hasattr(loop, "shutdown_asyncgens"):
254                      loop.run_until_complete(loop.shutdown_asyncgens())
255              finally:
256                  loop.close()
257                  asyncio.set_event_loop(self.main_event_loop)
258  
259      def __get__(self, parent, objtype):
260          """ 
261          Include self for methods 
262          """
263          func = functools.partial(self.__call__, parent)
264          return functools.update_wrapper(func, self.awaitable)

266      async def main_wrap(
267          self, args, kwargs, call_result, source_thread, exc_info, context
268      ):
269          """ 
270          Wraps the awaitable with something that puts the result into the 
271          result/exception future. 
272          """
273          if context is not None:
274              _restore_context(context[0])
275  
276          current_task = SyncToAsync.get_current_task()
277          self.launch_map[current_task] = source_thread
278          try:
279              # If we have an exception, run the function inside the except block
280              # after raising it so exc_info is correctly populated.
281              if exc_info[1]:
282                  try:
283                      raise exc_info[1]
284                  except BaseException:
285                      result = await self.awaitable(*args, **kwargs)
286              else:
287                  result = await self.awaitable(*args, **kwargs)
288          except BaseException as e:
289              call_result.set_exception(e)
290          else:
291              call_result.set_result(result)
292          finally:
293              del self.launch_map[current_task]
294  
295              if context is not None:
296                  context[0] = contextvars.copy_context()

299  class SyncToAsync:
300      """ 
301      Utility class which turns a synchronous callable into an awaitable that 
302      runs in a threadpool. It also sets a threadlocal inside the thread so 
303      calls to AsyncToSync can escape it. 
304   
305      If thread_sensitive is passed, the code will run in the same thread as any 
306      outer code. This is needed for underlying Python code that is not 
307      threadsafe (for example, code which handles SQLite database connections). 
308   
309      If the outermost program is async (i.e. SyncToAsync is outermost), then 
310      this will be a dedicated single sub-thread that all sync code runs in, 
311      one after the other. If the outermost program is sync (i.e. AsyncToSync is 
312      outermost), this will just be the main thread. This is achieved by idling 
313      with a CurrentThreadExecutor while AsyncToSync is blocking its sync parent, 
314      rather than just blocking. 
315   
316      If executor is passed in, that will be used instead of the loop's default executor. 
317      In order to pass in an executor, thread_sensitive must be set to False, otherwise 
318      a TypeError will be raised. 
319      """
320  
321      # If they've set ASGI_THREADS, update the default asyncio executor for now
322      if "ASGI_THREADS" in os.environ:
323          loop = asyncio.get_event_loop()
324          loop.set_default_executor(
325              ThreadPoolExecutor(max_workers=int(os.environ["ASGI_THREADS"]))
326          )
327  
328      # Maps launched threads to the coroutines that spawned them
329      launch_map: "Dict[threading.Thread, asyncio.Task[object]]" = {}
330  
331      # Storage for main event loop references
332      threadlocal = threading.local()
333  
334      # Single-thread executor for thread-sensitive code
335      single_thread_executor = ThreadPoolExecutor(max_workers=1)
336  
337      # Maintain a contextvar for the current execution context. Optionally used
338      # for thread sensitive mode.
339      if sys.version_info >= (3, 7):
340          thread_sensitive_context: "contextvars.ContextVar[str]" = (
341              contextvars.ContextVar("thread_sensitive_context")
342          )
343      else:
344          thread_sensitive_context: None = None
345  
346      # Maintaining a weak reference to the context ensures that thread pools are
347      # erased once the context goes out of scope. This terminates the thread pool.
348      context_to_thread_executor: "weakref.WeakKeyDictionary[object, ThreadPoolExecutor]" = (
349          weakref.WeakKeyDictionary()
350      )
351  
352      def __init__(
353          self,
354          func: Callable[..., Any],
355          thread_sensitive: bool = True,
356          executor: Optional["ThreadPoolExecutor"] = None,
357      ) -> None:
358          if not callable(func) or _iscoroutinefunction_or_partial(func):
359              raise TypeError("sync_to_async can only be applied to sync functions.")
360          self.func = func
361          functools.update_wrapper(self, func)
362          self._thread_sensitive = thread_sensitive
363          self._is_coroutine = asyncio.coroutines._is_coroutine  # type: ignore
364          if thread_sensitive and executor is not None:
365              raise TypeError("executor must not be set when thread_sensitive is True")
366          self._executor = executor
367          try:
368              self.__self__ = func.__self__  # type: ignore
369          except AttributeError:
370              pass
371  
372      async def __call__(self, *args, **kwargs):
373          loop = asyncio.get_event_loop()
374  
375          # Work out what thread to run the code in
376          if self._thread_sensitive:
377              if hasattr(AsyncToSync.executors, "current"):
378                  # If we have a parent sync thread above somewhere, use that
379                  executor = AsyncToSync.executors.current
380              elif self.thread_sensitive_context and self.thread_sensitive_context.get(
381                  None
382              ):
383                  # If we have a way of retrieving the current context, attempt
384                  # to use a per-context thread pool executor
385                  thread_sensitive_context = self.thread_sensitive_context.get()
386  
387                  if thread_sensitive_context in self.context_to_thread_executor:
388                      # Re-use thread executor in current context
389                      executor = self.context_to_thread_executor[thread_sensitive_context]
390                  else:
391                      # Create new thread executor in current context
392                      executor = ThreadPoolExecutor(max_workers=1)
393                      self.context_to_thread_executor[thread_sensitive_context] = executor
394              else:
395                  # Otherwise, we run it in a fixed single thread
396                  executor = self.single_thread_executor
397          else:
398              # Use the passed in executor, or the loop's default if it is None
399              executor = self._executor
400  
401          if contextvars is not None:
402              context = contextvars.copy_context()
403              child = functools.partial(self.func, *args, **kwargs)
404              func = context.run
405              args = (child,)
406              kwargs = {}
407          else:
408              func = self.func
409  
410          # Run the code in the right thread
411          future = loop.run_in_executor(
412              executor,
413              functools.partial(
414                  self.thread_handler,
415                  loop,
416                  self.get_current_task(),
417                  sys.exc_info(),
418                  func,
419                  *args,
420                  **kwargs,
421              ),
422          )
423          ret = await asyncio.wait_for(future, timeout=None)
424  
425          if contextvars is not None:
426              _restore_context(context)
427  
428          return ret

431          """ 
432          Include self for methods 
433          """
434          return functools.partial(self.__call__, parent)

436      def thread_handler(self, loop, source_task, exc_info, func, *args, **kwargs):
437          """ 
438          Wraps the sync application with exception handling. 
439          """
440          # Set the threadlocal for AsyncToSync
441          self.threadlocal.main_event_loop = loop
442          self.threadlocal.main_event_loop_pid = os.getpid()
443          # Set the task mapping (used for the locals module)
444          current_thread = threading.current_thread()
445          if AsyncToSync.launch_map.get(source_task) == current_thread:
446              # Our parent task was launched from this same thread, so don't make
447              # a launch map entry - let it shortcut over us! (and stop infinite loops)
448              parent_set = False
449          else:
450              self.launch_map[current_thread] = source_task
451              parent_set = True
452          # Run the function
453          try:
454              # If we have an exception, run the function inside the except block
455              # after raising it so exc_info is correctly populated.
456              if exc_info[1]:
457                  try:
458                      raise exc_info[1]
459                  except BaseException:
460                      return func(*args, **kwargs)
461              else:
462                  return func(*args, **kwargs)
463          finally:
464              # Only delete the launch_map parent if we set it, otherwise it is
465              # from someone else.
466              if parent_set:
467                  del self.launch_map[current_thread]

469      @staticmethod
470      def get_current_task():
471          """ 
472          Cross-version implementation of asyncio.current_task() 
473   
474          Returns None if there is no task. 
475          """
476          try:
477              if hasattr(asyncio, "current_task"):
478                  # Python 3.7 and up
479                  return asyncio.current_task()
480              else:
481                  # Python 3.6
482                  return asyncio.Task.current_task()
483          except RuntimeError:
484              return None

491  def sync_to_async(
492      func: Optional[Callable[..., Any]] = None,
493      thread_sensitive: bool = True,
494      executor: Optional["ThreadPoolExecutor"] = None,
495  ) -> Union[SyncToAsync, Callable[[Callable[..., Any]], SyncToAsync]]:
496      if func is None:
497          return lambda f: SyncToAsync(
498              f,
499              thread_sensitive=thread_sensitive,
500              executor=executor,
501          )
502      return SyncToAsync(
503          func,
504          thread_sensitive=thread_sensitive,
505          executor=executor,
506      )
