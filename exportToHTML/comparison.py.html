8    class Cast(Func):
9        """Coerce an expression to a new field type."""
10       function = 'CAST'
11       template = '%(function)s(%(expressions)s AS %(db_type)s)'
12   
13       def __init__(self, expression, output_field):
14           super().__init__(expression, output_field=output_field)
15   
16       def as_sql(self, compiler, connection, **extra_context):
17           extra_context['db_type'] = self.output_field.cast_db_type(connection)
18           return super().as_sql(compiler, connection, **extra_context)
19   
20       def as_sqlite(self, compiler, connection, **extra_context):
21           db_type = self.output_field.db_type(connection)
22           if db_type in {'datetime', 'time'}:
23               # Use strftime as datetime/time don't keep fractional seconds.
24               template = 'strftime(%%s, %(expressions)s)'
25               sql, params = super().as_sql(compiler, connection, template=template, **extra_context)
26               format_string = '%H:%M:%f' if db_type == 'time' else '%Y-%m-%d %H:%M:%f'
27               params.insert(0, format_string)
28               return sql, params
29           elif db_type == 'date':
30               template = 'date(%(expressions)s)'
31               return super().as_sql(compiler, connection, template=template, **extra_context)
32           return self.as_sql(compiler, connection, **extra_context)
33   
34       def as_mysql(self, compiler, connection, **extra_context):
35           template = None
36           output_type = self.output_field.get_internal_type()
37           # MySQL doesn't support explicit cast to float.
38           if output_type == 'FloatField':
39               template = '(%(expressions)s + 0.0)'
40           # MariaDB doesn't support explicit cast to JSON.
41           elif output_type == 'JSONField' and connection.mysql_is_mariadb:
42               template = "JSON_EXTRACT(%(expressions)s, '$')"
43           return self.as_sql(compiler, connection, template=template, **extra_context)
44   
45       def as_oracle(self, compiler, connection, **extra_context):
46           if self.output_field.get_internal_type() == 'JSONField':
47               # Oracle doesn't support explicit cast to JSON.
48               template = "JSON_QUERY(%(expressions)s, '$')"
49               return super().as_sql(compiler, connection, template=template, **extra_context)
50           return self.as_sql(compiler, connection, **extra_context)

53   class Coalesce(Func):
54       """Return, from left to right, the first non-null expression."""
55       function = 'COALESCE'
56   
57       def __init__(self, *expressions, **extra):
58           if len(expressions) < 2:
59               raise ValueError('Coalesce must take at least two expressions')
60           super().__init__(*expressions, **extra)
61   
62       def as_oracle(self, compiler, connection, **extra_context):
63           # Oracle prohibits mixing TextField (NCLOB) and CharField (NVARCHAR2),
64           # so convert all fields to NCLOB when that type is expected.
65           if self.output_field.get_internal_type() == 'TextField':
66               clone = self.copy()
67               clone.set_source_expressions([
68                   Func(expression, function='TO_NCLOB') for expression in self.get_source_expressions()
69               ])
70               return super(Coalesce, clone).as_sql(compiler, connection, **extra_context)
71           return self.as_sql(compiler, connection, **extra_context)

74   class Collate(Func):
75       function = 'COLLATE'
76       template = '%(expressions)s %(function)s %(collation)s'
77       # Inspired from https://www.postgresql.org/docs/current/sql-syntax-lexical.html#SQL-SYNTAX-IDENTIFIERS
78       collation_re = _lazy_re_compile(r'^[\w\-]+$')
79   
80       def __init__(self, expression, collation):
81           if not (collation and self.collation_re.match(collation)):
82               raise ValueError('Invalid collation name: %r.' % collation)
83           self.collation = collation
84           super().__init__(expression)
85   
86       def as_sql(self, compiler, connection, **extra_context):
87           extra_context.setdefault('collation', connection.ops.quote_name(self.collation))
88           return super().as_sql(compiler, connection, **extra_context)

91   class Greatest(Func):
92       """ 
93       Return the maximum expression. 
94    
95       If any expression is null the return value is database-specific: 
96       On PostgreSQL, the maximum not-null expression is returned. 
97       On MySQL, Oracle, and SQLite, if any expression is null, null is returned. 
98       """
99       function = 'GREATEST'
100  
101      def __init__(self, *expressions, **extra):
102          if len(expressions) < 2:
103              raise ValueError('Greatest must take at least two expressions')
104          super().__init__(*expressions, **extra)
105  
106      def as_sqlite(self, compiler, connection, **extra_context):
107          """Use the MAX function on SQLite."""
108          return super().as_sqlite(compiler, connection, function='MAX', **extra_context)

111  class JSONObject(Func):
112      function = 'JSON_OBJECT'
113      output_field = JSONField()
114  
115      def __init__(self, **fields):
116          expressions = []
117          for key, value in fields.items():
118              expressions.extend((Value(key), value))
119          super().__init__(*expressions)
120  
121      def as_sql(self, compiler, connection, **extra_context):
122          if not connection.features.has_json_object_function:
123              raise NotSupportedError(
124                  'JSONObject() is not supported on this database backend.'
125              )
126          return super().as_sql(compiler, connection, **extra_context)
127  
128      def as_postgresql(self, compiler, connection, **extra_context):
129          return self.as_sql(
130              compiler,
131              connection,
132              function='JSONB_BUILD_OBJECT',
133              **extra_context,
134          )
135  
136      def as_oracle(self, compiler, connection, **extra_context):
137          class ArgJoiner:
138              def join(self, args):
139                  args = [' VALUE '.join(arg) for arg in zip(args[::2], args[1::2])]
140                  return ', '.join(args)
141  
142          return self.as_sql(
143              compiler,
144              connection,
145              arg_joiner=ArgJoiner(),
146              template='%(function)s(%(expressions)s RETURNING CLOB)',
147              **extra_context,
148          )

171  class NullIf(Func):
172      function = 'NULLIF'
173      arity = 2
174  
175      def as_oracle(self, compiler, connection, **extra_context):
176          expression1 = self.get_source_expressions()[0]
177          if isinstance(expression1, Value) and expression1.value is None:
178              raise ValueError('Oracle does not allow Value(None) for expression1.')
179          return super().as_sql(compiler, connection, **extra_context)
