20   def supports_color():
21       """ 
22       Return True if the running system's terminal supports color, 
23       and False otherwise. 
24       """
25       def vt_codes_enabled_in_windows_registry():
26           """ 
27           Check the Windows Registry to see if VT code handling has been enabled 
28           by default, see https://superuser.com/a/1300251/447564. 
29           """
30           try:
31               # winreg is only available on Windows.
32               import winreg
33           except ImportError:
34               return False
35           else:
36               reg_key = winreg.OpenKey(winreg.HKEY_CURRENT_USER, 'Console')
37               try:
38                   reg_key_value, _ = winreg.QueryValueEx(reg_key, 'VirtualTerminalLevel')
39               except FileNotFoundError:
40                   return False
41               else:
42                   return reg_key_value == 1
43   
44       # isatty is not always implemented, #6223.
45       is_a_tty = hasattr(sys.stdout, 'isatty') and sys.stdout.isatty()
46   
47       return is_a_tty and (
48           sys.platform != 'win32' or
49           HAS_COLORAMA or
50           'ANSICON' in os.environ or
51           # Windows Terminal supports VT codes.
52           'WT_SESSION' in os.environ or
53           # Microsoft Visual Studio Code's built-in terminal supports colors.
54           os.environ.get('TERM_PROGRAM') == 'vscode' or
55           vt_codes_enabled_in_windows_registry()
56       )

60       pass

63   def make_style(config_string=''):
64       """ 
65       Create a Style object from the given config_string. 
66    
67       If config_string is empty django.utils.termcolors.DEFAULT_PALETTE is used. 
68       """
69   
70       style = Style()
71   
72       color_settings = termcolors.parse_color_setting(config_string)
73   
74       # The nocolor palette has all available roles.
75       # Use that palette as the basis for populating
76       # the palette as defined in the environment.
77       for role in termcolors.PALETTES[termcolors.NOCOLOR_PALETTE]:
78           if color_settings:
79               format = color_settings.get(role, {})
80               style_func = termcolors.make_style(**format)
81           else:
82               def style_func(x):
83                   return x
84           setattr(style, role, style_func)
85   
86       # For backwards compatibility,
87       # set style for ERROR_OUTPUT == ERROR
88       style.ERROR_OUTPUT = style.ERROR
89   
90       return style

93   @functools.lru_cache(maxsize=None)
94   def no_style():
95       """ 
96       Return a Style object with no color scheme. 
97       """
98       return make_style('nocolor')

101  def color_style(force_color=False):
102      """ 
103      Return a Style object from the Django color scheme. 
104      """
105      if not force_color and not supports_color():
106          return no_style()
107      return make_style(os.environ.get('DJANGO_COLORS', ''))
