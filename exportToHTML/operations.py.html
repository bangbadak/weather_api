9    class CreateExtension(Operation):
10       reversible = True
11   
12       def __init__(self, name):
13           self.name = name
14   
15       def state_forwards(self, app_label, state):
16           pass
17   
18       def database_forwards(self, app_label, schema_editor, from_state, to_state):
19           if (
20               schema_editor.connection.vendor != 'postgresql' or
21               not router.allow_migrate(schema_editor.connection.alias, app_label)
22           ):
23               return
24           if not self.extension_exists(schema_editor, self.name):
25               schema_editor.execute(
26                   'CREATE EXTENSION IF NOT EXISTS %s' % schema_editor.quote_name(self.name)
27               )
28           # Clear cached, stale oids.
29           get_hstore_oids.cache_clear()
30           get_citext_oids.cache_clear()
31           # Registering new type handlers cannot be done before the extension is
32           # installed, otherwise a subsequent data migration would use the same
33           # connection.
34           register_type_handlers(schema_editor.connection)
35   
36       def database_backwards(self, app_label, schema_editor, from_state, to_state):
37           if not router.allow_migrate(schema_editor.connection.alias, app_label):
38               return
39           if self.extension_exists(schema_editor, self.name):
40               schema_editor.execute(
41                   'DROP EXTENSION IF EXISTS %s' % schema_editor.quote_name(self.name)
42               )
43           # Clear cached, stale oids.
44           get_hstore_oids.cache_clear()
45           get_citext_oids.cache_clear()
46   
47       def extension_exists(self, schema_editor, extension):
48           with schema_editor.connection.cursor() as cursor:
49               cursor.execute(
50                   'SELECT 1 FROM pg_extension WHERE extname = %s',
51                   [extension],
52               )
53               return bool(cursor.fetchone())
54   
55       def describe(self):
56           return "Creates extension %s" % self.name
57   
58       @property
59       def migration_name_fragment(self):
60           return 'create_extension_%s' % self.name

111  class NotInTransactionMixin:
112      def _ensure_not_in_transaction(self, schema_editor):
113          if schema_editor.connection.in_atomic_block:
114              raise NotSupportedError(
115                  'The %s operation cannot be executed inside a transaction '
116                  '(set atomic = False on the migration).'
117                  % self.__class__.__name__
118              )

121  class AddIndexConcurrently(NotInTransactionMixin, AddIndex):
122      """Create an index using PostgreSQL's CREATE INDEX CONCURRENTLY syntax."""
123      atomic = False
124  
125      def describe(self):
126          return 'Concurrently create index %s on field(s) %s of model %s' % (
127              self.index.name,
128              ', '.join(self.index.fields),
129              self.model_name,
130          )
131  
132      def database_forwards(self, app_label, schema_editor, from_state, to_state):
133          self._ensure_not_in_transaction(schema_editor)
134          model = to_state.apps.get_model(app_label, self.model_name)
135          if self.allow_migrate_model(schema_editor.connection.alias, model):
136              schema_editor.add_index(model, self.index, concurrently=True)
137  
138      def database_backwards(self, app_label, schema_editor, from_state, to_state):
139          self._ensure_not_in_transaction(schema_editor)
140          model = from_state.apps.get_model(app_label, self.model_name)
141          if self.allow_migrate_model(schema_editor.connection.alias, model):
142              schema_editor.remove_index(model, self.index, concurrently=True)

145  class RemoveIndexConcurrently(NotInTransactionMixin, RemoveIndex):
146      """Remove an index using PostgreSQL's DROP INDEX CONCURRENTLY syntax."""
147      atomic = False
148  
149      def describe(self):
150          return 'Concurrently remove index %s from %s' % (self.name, self.model_name)
151  
152      def database_forwards(self, app_label, schema_editor, from_state, to_state):
153          self._ensure_not_in_transaction(schema_editor)
154          model = from_state.apps.get_model(app_label, self.model_name)
155          if self.allow_migrate_model(schema_editor.connection.alias, model):
156              from_model_state = from_state.models[app_label, self.model_name_lower]
157              index = from_model_state.get_index_by_name(self.name)
158              schema_editor.remove_index(model, index, concurrently=True)
159  
160      def database_backwards(self, app_label, schema_editor, from_state, to_state):
161          self._ensure_not_in_transaction(schema_editor)
162          model = to_state.apps.get_model(app_label, self.model_name)
163          if self.allow_migrate_model(schema_editor.connection.alias, model):
164              to_model_state = to_state.models[app_label, self.model_name_lower]
165              index = to_model_state.get_index_by_name(self.name)
166              schema_editor.add_index(model, index, concurrently=True)

169  class CollationOperation(Operation):
170      def __init__(self, name, locale, *, provider='libc', deterministic=True):
171          self.name = name
172          self.locale = locale
173          self.provider = provider
174          self.deterministic = deterministic
175  
176      def state_forwards(self, app_label, state):
177          pass
178  
179      def deconstruct(self):
180          kwargs = {'name': self.name, 'locale': self.locale}
181          if self.provider and self.provider != 'libc':
182              kwargs['provider'] = self.provider
183          if self.deterministic is False:
184              kwargs['deterministic'] = self.deterministic
185          return (
186              self.__class__.__qualname__,
187              [],
188              kwargs,
189          )
190  
191      def create_collation(self, schema_editor):
192          if (
193              self.deterministic is False and
194              not schema_editor.connection.features.supports_non_deterministic_collations
195          ):
196              raise NotSupportedError(
197                  'Non-deterministic collations require PostgreSQL 12+.'
198              )
199          if (
200              self.provider != 'libc' and
201              not schema_editor.connection.features.supports_alternate_collation_providers
202          ):
203              raise NotSupportedError('Non-libc providers require PostgreSQL 10+.')
204          args = {'locale': schema_editor.quote_name(self.locale)}
205          if self.provider != 'libc':
206              args['provider'] = schema_editor.quote_name(self.provider)
207          if self.deterministic is False:
208              args['deterministic'] = 'false'
209          schema_editor.execute('CREATE COLLATION %(name)s (%(args)s)' % {
210              'name': schema_editor.quote_name(self.name),
211              'args': ', '.join(f'{option}={value}' for option, value in args.items()),
212          })
213  
214      def remove_collation(self, schema_editor):
215          schema_editor.execute(
216              'DROP COLLATION %s' % schema_editor.quote_name(self.name),
217          )

221      """Create a collation."""
222      def database_forwards(self, app_label, schema_editor, from_state, to_state):
223          if (
224              schema_editor.connection.vendor != 'postgresql' or
225              not router.allow_migrate(schema_editor.connection.alias, app_label)
226          ):

234  
235      def describe(self):
